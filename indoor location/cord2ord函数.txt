/*
    输入节点坐标，输出相连的节点序号
*/
#include <stdio.h>
#include <math.h>
#define MAX_VERTEX 9 //顶点个数

typedef struct point{
    int x;
    int y;
    int flag;       //障碍物flag = 1，否则flag = 0
};

void cord2ord(struct point *p);

struct point point[MAX_VERTEX] = {
                                  {0, 0, 0}, {1, 0, 0}, {2, 0, 0},
                                  {0, 1, 0}, {1, 1, 1}, {2, 1, 0},
                                  {0, 2, 0}, {1, 2, 0}, {2, 2, 0}
                                                                };

int main(){
    cord2ord(&point);
    return 0;
}

void cord2ord(struct point *p){
    int i, j;

    int edge  = (int)sqrt(MAX_VERTEX);
    for(i = 0; i < edge; i++){
        for(j = 0; j < edge; j++){
            //输出相连上顶点
            if(p[j * edge + i].flag == 0 && p[(j + 1) * edge + i].flag == 0  && (j >= 0 && j <= edge - 2)){
                printf("%d %d\n", j * edge + i, (j + 1) * edge + i);
            }
            //输出相连右顶点
            if(p[j * edge + i].flag == 0 && p[j * edge + i + 1].flag == 0 && (i >= 0 && i <= edge - 2)){
                printf("%d %d\n", j * edge + i, j * edge + i + 1);
            }
        }
    }
}
